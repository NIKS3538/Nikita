#import mplot3d toolkits,numpy and matplotlib
from mpl_toolkits import mplot3d
import numpy as np
import matplotlib.pyplot as plt

fig=plt.figure()
#syntax for 3-Dcprojection
ax=plt.axes(projection='3d')

#define all 3 axes
z=np.linspace(0,1,100)
x=z*np.sin(25*z)
y=z*np.cos(25*z)

#plotting
ax.plot3D(x,y,z,'green')
ax.set_title('3D line plot')
plt.show()
